 import scipy

 sklearn.datasets import load_breast_cancer 
 data = load_breast_cancer ( ) 
 X = data.data # features 
 y = data.target # labels

 n_bootstrap = 100 

 # Define list store bootstrap samples

 range(n_bootstrap ): 
     # Draw random indices replacement 
     indices = np.random.choice(len(X ) , size = len(X ) , replace = True ) 
     # Extract bootstrap sample 
     X_bootstrap = X[indices ] 
     y_bootstrap = y[indices ] 
     # Append bootstrap sample list 
     bootstrap_samples.append((X_bootstrap , y_bootstrap ) ) 

 # Train logistic regression model training set 
 sklearn.linear_model import LogisticRegression 
 model_A = LogisticRegression ( ) 
 model_A.fit(X_train , y_train ) 

 # Train support vector machine model training set 
 sklearn.svm import SVC 
 model_B = SVC ( ) 
 model_B.fit(X_train , y_train ) 

 # Define list store bootstrap estimates 
 bootstrap_estimates_A = [ ] 
 bootstrap_estimates_B = [ ] 

 # Loop bootstrap samples 
 X_bootstrap , y_bootstrap bootstrap_samples : 
     # Evaluate models bootstrap sample accuracy metric 
     y_pred_A = model_A.predict(X_bootstrap ) 
     accuracy_A = sklearn.metrics.accuracy_score(y_bootstrap , y_pred_A ) 
     y_pred_B = model_B.predict(X_bootstrap ) 
     accuracy_B = sklearn.metrics.accuracy_score(y_bootstrap , y_pred_B ) 
     # Append accuracies lists 
     bootstrap_estimates_A.append(accuracy_A ) 
     bootstrap_estimates_B.append(accuracy_B ) 

 # Perform - sample test test mean accuracy model equal 0.8 
 # Use scipy.stats.ttest_1samp function 
 t_stat , p_value = scipy.stats.ttest_1samp(bootstrap_estimates_A , 0.8 )

 # Perform - sample test test mean accuracy model equal mean accuracy model B 
 # Use scipy.stats.ttest_ind function
